"""
Chemical Compliance Checker
Lightweight molecular validation using RDKit (no Docker required)
"""
from typing import Dict, List, Optional
from dataclasses import dataclass, field
from enum import Enum

from rdkit import Chem
from rdkit.Chem import Descriptors, Crippen, Lipinski, rdMolDescriptors
from rdkit.Chem.FilterCatalog import FilterCatalog, FilterCatalogParams


class ComplianceLevel(str, Enum):
    PASS = "PASS"
    WARNING = "WARNING"
    FAIL = "FAIL"


@dataclass
class ComplianceIssue:
    category: str
    level: ComplianceLevel
    message: str
    details: Optional[Dict] = None


@dataclass
class ComplianceReport:
    overall_status: ComplianceLevel = ComplianceLevel.PASS
    issues: List[ComplianceIssue] = field(default_factory=list)
    passed_checks: List[str] = field(default_factory=list)
    metadata: Dict = field(default_factory=dict)
    
    def add_issue(self, category: str, level: ComplianceLevel, message: str, details: Dict = None):
        self.issues.append(ComplianceIssue(category, level, message, details))
        if level == ComplianceLevel.FAIL:
            self.overall_status = ComplianceLevel.FAIL
        elif level == ComplianceLevel.WARNING and self.overall_status != ComplianceLevel.FAIL:
            self.overall_status = ComplianceLevel.WARNING
    
    def add_pass(self, check_name: str):
        self.passed_checks.append(check_name)
    
    def to_dict(self) -> Dict:
        return {
            "overall_status": self.overall_status.value,
            "passed_checks": self.passed_checks,
            "issues": [
                {
                    "category": issue.category,
                    "level": issue.level.value,
                    "message": issue.message,
                    "details": issue.details
                }
                for issue in self.issues
            ],
            "metadata": self.metadata
        }


class MolecularComplianceChecker:
    """
    Lightweight molecular compliance checker using RDKit.
    No Docker required - runs directly in Python environment.
    """
    
    def __init__(self):
        # Initialize PAINS filter catalog
        params = FilterCatalogParams()
        params.AddCatalog(FilterCatalogParams.FilterCatalogs.PAINS_A)
        self.pains_catalog = FilterCatalog(params)
        
        # Initialize Brenk filter catalog (reactive groups)
        params_brenk = FilterCatalogParams()
        params_brenk.AddCatalog(FilterCatalogParams.FilterCatalogs.BRENK)
        self.brenk_catalog = FilterCatalog(params_brenk)
    
    def check_molecule(self, smiles: str) -> ComplianceReport:
        """
        Check a single molecule by SMILES.
        
        Args:
            smiles: SMILES string
            
        Returns:
            ComplianceReport with validation results
        """
        report = ComplianceReport()
        report.metadata["smiles"] = smiles
        
        # Parse molecule
        mol = Chem.MolFromSmiles(smiles)
        if mol is None:
            report.add_issue(
                category="validity",
                level=ComplianceLevel.FAIL,
                message="Invalid SMILES string",
                details={"smiles": smiles}
            )
            return report
        
        # Run all checks
        self._check_pains(mol, report)
        self._check_brenk(mol, report)
        self._check_lipinski(mol, report)
        self._check_toxicity_alerts(mol, report)
        self._check_structure_properties(mol, report)
        
        return report
    
    def check_sdf_file(self, sdf_path: str) -> List[ComplianceReport]:
        """
        Check all molecules in an SDF file.
        
        Args:
            sdf_path: Path to SDF file
            
        Returns:
            List of ComplianceReport for each molecule
        """
        supplier = Chem.SDMolSupplier(sdf_path)
        reports = []
        
        for i, mol in enumerate(supplier):
            if mol is None:
                reports.append(ComplianceReport(
                    overall_status=ComplianceLevel.FAIL,
                    issues=[ComplianceIssue(
                        category="validity",
                        level=ComplianceLevel.FAIL,
                        message=f"Molecule {i} could not be parsed"
                    )]
                ))
                continue
            
            smiles = mol.GetProp("_Name") if mol.HasProp("_Name") else f"mol_{i}"
            report = self.check_molecule_from_rdkit(mol, smiles)
            reports.append(report)
        
        return reports
    
    def check_molecule_from_rdkit(self, mol: Chem.Mol, name: str = "") -> ComplianceReport:
        """Check a molecule from RDKit Mol object."""
        smiles = name or Chem.MolToSmiles(mol)
        report = ComplianceReport()
        report.metadata["smiles"] = smiles
        
        self._check_pains(mol, report)
        self._check_brenk(mol, report)
        self._check_lipinski(mol, report)
        self._check_toxicity_alerts(mol, report)
        self._check_structure_properties(mol, report)
        
        return report
    
    def _check_pains(self, mol: Chem.Mol, report: ComplianceReport):
        """Check for PAINS (Pan-Assay Interference Compounds)."""
        has_pains = self.pains_catalog.HasMatch(mol)
        
        if has_pains:
            matches = list(self.pains_catalog.GetMatches(mol))
            report.add_issue(
                category="pains_filter",
                level=ComplianceLevel.FAIL,
                message="Contains PAINS substructure (likely false positive)",
                details={"num_matches": len(matches)}
            )
        else:
            report.add_pass("pains_filter")
    
    def _check_brenk(self, mol: Chem.Mol, report: ComplianceReport):
        """Check for reactive/unstable groups (Brenk filter)."""
        has_brenk = self.brenk_catalog.HasMatch(mol)
        
        if has_brenk:
            matches = list(self.brenk_catalog.GetMatches(mol))
            report.add_issue(
                category="brenk_filter",
                level=ComplianceLevel.WARNING,
                message="Contains reactive or unstable functional groups",
                details={"num_matches": len(matches)}
            )
        else:
            report.add_pass("brenk_filter")
    
    def _check_lipinski(self, mol: Chem.Mol, report: ComplianceReport):
        """Check Lipinski's Rule of Five."""
        mw = Descriptors.MolWt(mol)
        logp = Crippen.MolLogP(mol)
        hbd = Lipinski.NumHDonors(mol)
        hba = Lipinski.NumHAcceptors(mol)
        
        violations = []
        if mw > 500:
            violations.append(f"Molecular weight {mw:.2f} > 500")
        if logp > 5:
            violations.append(f"LogP {logp:.2f} > 5")
        if hbd > 5:
            violations.append(f"H-bond donors {hbd} > 5")
        if hba > 10:
            violations.append(f"H-bond acceptors {hba} > 10")
        
        report.metadata.update({
            "molecular_weight": round(mw, 2),
            "logp": round(logp, 2),
            "hbd": hbd,
            "hba": hba
        })
        
        if violations:
            report.add_issue(
                category="lipinski_rule_of_5",
                level=ComplianceLevel.WARNING,
                message=f"Violates {len(violations)} Lipinski rule(s)",
                details={"violations": violations}
            )
        else:
            report.add_pass("lipinski_rule_of_5")
    
    def _check_toxicity_alerts(self, mol: Chem.Mol, report: ComplianceReport):
        """Check for common toxicity alerts."""
        alerts = []
        
        # Nitro groups
        if Chem.SubstructMatch(mol, Chem.MolFromSmarts("[N+](=O)[O-]")):
            alerts.append("Nitro group detected (potential genotoxicity)")
        
        # Aldehydes
        if Chem.SubstructMatch(mol, Chem.MolFromSmarts("[CX3H1]=O")):
            alerts.append("Aldehyde group detected (reactive)")
        
        # Michael acceptors
        if Chem.SubstructMatch(mol, Chem.MolFromSmarts("[C,C]=[C,C][C,c]=[O,N,S]")):
            alerts.append("Potential Michael acceptor (non-specific alkylation)")
        
        # Epoxides
        if Chem.SubstructMatch(mol, Chem.MolFromSmarts("C1CO1")):
            alerts.append("Epoxide ring detected (highly reactive)")
        
        if alerts:
            report.add_issue(
                category="toxicity_alerts",
                level=ComplianceLevel.WARNING,
                message="Potential toxicity concerns",
                details={"alerts": alerts}
            )
        else:
            report.add_pass("toxicity_alerts")
    
    def _check_structure_properties(self, mol: Chem.Mol, report: ComplianceReport):
        """Check basic structural properties."""
        alerts = []
        
        mw = Descriptors.MolWt(mol)
        if mw < 150:
            alerts.append(f"Molecular weight too low: {mw:.2f}")
        elif mw > 1000:
            alerts.append(f"Molecular weight too high: {mw:.2f}")
        
        rot_bonds = Descriptors.NumRotatableBonds(mol)
        if rot_bonds > 15:
            alerts.append(f"Too many rotatable bonds: {rot_bonds}")
        
        arom_rings = rdMolDescriptors.CalcNumAromaticRings(mol)
        if arom_rings > 5:
            alerts.append(f"Too many aromatic rings: {arom_rings}")
        
        if alerts:
            report.add_issue(
                category="structure_alerts",
                level=ComplianceLevel.WARNING,
                message="Structural concerns",
                details={"alerts": alerts}
            )
        else:
            report.add_pass("structure_properties")

Pythonclass IndustrialComplianceChecker:    """    工业级分子合规检查器    整合预处理 + 全量检查 + 评分 + 批处理    """        def __init__(self, n_workers: int = 4):        self.preprocessor = MoleculePreprocessor()        self.checks       = ComplianceChecks()        self.n_workers    = n_workers        def check_molecule(        self,        smiles     : str,        mol_id     : str = "",        preprocess : bool = True    ) -> ComplianceReport:                report            = ComplianceReport()        report.molecule_id = mol_id or smiles[:20]        report.metadata["input_smiles"] = smiles                # def _validate_pdb_format(file_path: str) -> Dict:
    """Validate PDB file format."""
    issues = []
    warnings = []
    
    try:
        with open(file_path, 'r') as f:
            lines = f.readlines()
        
        has_atom = False
        atom_count = 0
        
        for line in lines:
            if line.startswith("ATOM") or line.startswith("HETATM"):
                has_atom = True
                atom_count += 1
                
                # Basic format check
                if len(line) < 54:
                    issues.append(f"Invalid ATOM line format at line {atom_count}")
        
        if not has_atom:
            issues.append("No ATOM records found")
        
        if atom_count == 0:
            issues.append("Empty structure")
        
        return {
            "valid": len(issues) == 0,
            "issues": issues,
            "warnings": warnings,
            "atom_count": atom_count
        }
        
    except Exception as e:
        return {
            "valid": False,
            "issues": [str(e)],
            "warnings": [],
            "atom_count": 0
        }

Python# app/services/checker.pyimport loggingfrom typing import Dict, List, Optionalfrom concurrent.futures import ThreadPoolExecutor, as_completedfrom rdkit import Chemfrom app.models.schemas import ComplianceReport, ComplianceLevel, CheckCategoryfrom app.services.preprocessor import MoleculePreprocessorfrom app.services.checks import ComplianceCheckslogger = logging.getLogger(__name__)class IndustrialComplianceChecker:    """    工业级分子合规检查器（单例模式）    """    _instance = None    def __new__(cls):        if cls._instance is None:            cls._instance = super().__new__(cls)            cls._instance._initialized = False        return cls._instance    def __init__(self):        if self._initialized:            return        self.preprocessor  = MoleculePreprocessor()        self.checks        = ComplianceChecks()        self._initialized  = True        logger.info("IndustrialComplianceChecker 初始化完成")    # ----------------------------------------------------------    # 单分子检查    # ----------------------------------------------------------    def check_molecule(        self,        smiles     : str,        mol_id     : str  = "",        preprocess : bool = True    ) -> ComplianceReport:        report             = ComplianceReport()        report.molecule_id = mol_id or smiles[:30]        report.metadata["input_smiles"] = smiles        # 解析SMILES        mol = Chem.MolFromSmiles(smiles)        if mol is None:            report.add_issue(                category = CheckCategory.VALIDITY,                level    = ComplianceLevel.FAIL,                message  = "无效的SMILES字符串，无法解析",                details  = {"input": smiles},                penalty  = 100.0            )            return report        report.add_pass(CheckCategory.VALIDITY)        # 预处理        if preprocess:            mol = self.preprocessor.preprocess(mol, report)            if mol is None:                return report        # 按顺序执行所有检查        check_pipeline = [            ("元素检查",     self.checks.check_elements),            ("PAINS检查",    self.checks.check_pains),            ("Brenk检查",    self.checks.check_brenk),            ("成药性检查",   self.checks.check_drug_likeness),            ("毒性警报",     self.checks.check_toxicity_alerts),            ("hERG检查",     self.checks.check_herg_risk),            ("合成可及性",   self.checks.check_synthetic_accessibility),        ]        for name, fn in check_pipeline:            try:                fn(mol, report)            except Exception as e:                logger.warning(f"检查模块[{name}]异常: {e}")                report.add_issue(                    category = "system",                    level    = ComplianceLevel.WARNING,                    message  = f"检查模块[{name}]执行异常: {str(e)}"                )        return report    # ----------------------------------------------------------    # 批量检查（多线程）    # ----------------------------------------------------------    def check_batch(        self,        smiles_list : List[str],        mol_ids     : Optional[List[str]] = None,        preprocess  : bool                = True,        n_workers   : int                 = 4    ) -> List[ComplianceReport]:        if mol_ids is None:            mol_ids = [f"mol_{i}" for i in range(len(smiles_list))]        results = [None] * len(smiles_list)        with ThreadPoolExecutor(max_workers=n_workers) as executor:            futures = {                executor.submit(                    self.check_molecule, smi, mid, preprocess                ): idx                for idx, (smi, mid) in enumerate(zip(smiles_list, mol_ids))            }            for future in as_completed(futures):                idx         = futures[future]                results[idx] = future.result()        return results    # ----------------------------------------------------------    # 汇总统计    # ----------------------------------------------------------    def generate_summary(        self, reports: List[ComplianceReport]    ) -> Dict:        total   = len(reports)        passed  = sum(1 for r in reports                      if r.overall_status == ComplianceLevel.PASS)        warning = sum(1 for r in reports                      if r.overall_status == ComplianceLevel.WARNING)        failed  = sum(1 for r in reports                      if r.overall_status == ComplianceLevel.FAIL)        avg_score = sum(r.score for r in reports) / total if total else 0        # 统计最常见失败原因        counter = {}        for r in reports:            for issue in r.issues:                if issue.level == ComplianceLevel.FAIL:                    counter[issue.category] = counter.get(                        issue.category, 0) + 1        top_failures = sorted(            [{"category": k, "count": v} for k, v in counter.items()],            key=lambda x: x["count"],            reverse=True        )[:5]        return {            "total"       : total,            "passed"      : passed,            "warnings"    : warning,            "failed"      : failed,            "pass_rate"   : f"{passed/total*100:.1f}%" if total else "N/A",            "avg_score"   : round(avg_score, 2),            "top_failures": top_failures        }

Python# checker.pyimport loggingfrom typing import List, Optionalfrom rdkit import Chemfrom models import ComplianceReport, ComplianceLevel, CheckCategoryfrom preprocessor import MoleculePreprocessorfrom checks import ComplianceCheckslogger = logging.getLogger(__name__)class MolecularComplianceChecker:    def __init__(self):        self.preprocessor = MoleculePreprocessor()        self.checks       = ComplianceChecks()        logger.info("MolecularComplianceChecker 初始化完成")    def check_molecule(        self,        smiles     : str,        mol_id     : str  = "",        preprocess : bool = True    ) -> ComplianceReport:        report             = ComplianceReport()        report.molecule_id = mol_id or smiles[:30]        report.metadata["input_smiles"] = smiles        # Step 1: 解析SMILES        mol = Chem.MolFromSmiles(smiles)        if mol is None:            report.add_issue(                category = CheckCategory.VALIDITY,                level    = ComplianceLevel.FAIL,                message  = "无效的SMILES字符串，无法解析",                details  = {"input": smiles},                penalty  = 100.0            )            return report        report.add_pass(CheckCategory.VALIDITY)        # Step 2: 预处理（可选）        if preprocess:            mol = self.preprocessor.preprocess(mol, report)            if mol is None:                return report        # Step 3: 执行所有检查        check_pipeline = [            ("元素检查",   self.checks.check_elements),            ("PAINS检查",  self.checks.check_pains),            ("Brenk检查",  self.checks.check_brenk),            ("成药性检查", self.checks.check_drug_likeness),            ("毒性警报",   self.checks.check_toxicity_alerts),            ("hERG检查",   self.checks.check_herg_risk),            ("合成可及性", self.checks.check_synthetic_accessibility),        ]        for name, fn in check_pipeline:            try:                fn(mol, report)            except Exception as e:                logger.warning(f"检查模块[{name}]异常: {e}")                report.add_issue(                    category = "system",                    level    = ComplianceLevel.WARNING,                    message  = f"检查模块[{name}]执行异常: {str(e)}"                )        return report    def check_batch(        self,        smiles_list : List[str],        mol_ids     : Optional[List[str]] = None,        preprocess  : bool                = True    ) -> List[ComplianceReport]:        if mol_ids is None:            mol_ids = [f"mol_{i}" for i in range(len(smiles_list))]        return [            self.check_molecule(smi, mid, preprocess)            for smi, mid in zip(smiles_list, mol_ids)        ]

理解异步适用场景
代码异步真正发挥作用的场景：┌─────────────────────────────────────────────────────┐│  I/O 密集型（异步有效）                              ││                                                     ││  等待数据库查询   ──► 等待期间可以处理其他请求       ││  等待外部API响应  ──► 等待期间可以处理其他请求       ││  等待文件读写     ──► 等待期间可以处理其他请求       ││                                                     ││  特点：CPU空闲在等待，异步让等待时间被充分利用       │└─────────────────────────────────────────────────────┘┌─────────────────────────────────────────────────────┐│  CPU 密集型（异步无效）                              ││                                                     ││  RDKit 分子解析      ──► CPU 一直在计算             ││  PAINS 子结构匹配    ──► CPU 一直在计算             ││  描述符计算          ──► CPU 一直在计算             ││  互变异构体标准化    ──► CPU 一直在计算             ││                                                     ││  特点：CPU 始终满载，没有等待，异步没有优化空间      │└─────────────────────────────────────────────────────┘
